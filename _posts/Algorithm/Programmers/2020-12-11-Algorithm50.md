# **코드**

- C++ 코드

```cpp
#include <vector>

using namespace std;

int solution(int _num) {
    long long num = _num;
    int answer = 0;
    while (1){
        if(num == 1){
            break;
        }
        answer++;
        if(num % 2 == 0){
            num /= 2;
        }
        else{
            num = (num*3) + 1;
        }
        if(answer == 500){
            return -1;
        }
    }
    return answer;
}
```
